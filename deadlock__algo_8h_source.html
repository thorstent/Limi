81  deadlock_result<Symbol,State> run(const Automaton& automaton)
82  {
83 
84  std::unordered_set<State> initial_states = automaton.initial_states();
85  std::unordered_set<State> reachable; // a list of reachable states
86  std::deque<frontier_item> frontier = initial_states_conv(initial_states);
87  std::unordered_map<State,std::unordered_set<State>> predecessor_map;
88  std::unordered_set<State> final_states;
89 
90  deadlock_result<Symbol,State> result;
91  result.deadlock_found = false;
92 
93 #ifdef DEBUG_PRINTING
94  unsigned loop_counter = 0;
95 #endif
96  while (frontier.size() > 0) {
97 #ifdef DEBUG_PRINTING
98  ++loop_counter;
99  if (DEBUG_PRINTING>=2 && loop_counter % 100 == 0) std::cout << loop_counter << " rounds; seen states: " << reachable.size() << std::endl;
100 #endif
101  const frontier_item current = frontier.front();
102  frontier.pop_front();
103 #ifdef DEBUG_PRINTING
104  if (DEBUG_PRINTING>=3) std::cout << automaton.state_printer()(current.s) << std::endl;
105 #endif
106 
107  if (reachable.insert(current.s).second) {
108  if (automaton.is_final_state(current.s))
109  final_states.insert(current.s);
110  bool has_successors = false;
111  bool one_assumes = false;
112  Symbol_set next_symbols = automaton.next_symbols(current.s);
113  for (Symbol sigma : next_symbols) {
114  // TODO Fix this (piercing throuh the template)
115  one_assumes = one_assumes || sigma.symbol->assume;
116  State_set states = automaton.successors(current.s, sigma);
117 
118  for (State state : states) {
119  frontier_item next(state, current.cex_chain, sigma);
120  frontier.push_back(next);
121  has_successors = true;
122  predecessor_map[state].insert(current.s);
123  }
124 
125  }
126 
127  if (!automaton.is_final_state(current.s) && !has_successors) {
128  if (one_assumes) {
129  // add this to the final states because reaching this state is ok as well
130  final_states.insert(current.s);
131  } else {
132  result.counter_example = current.cex_chain->to_list();
133  result.deadlock_found = true;
134  result.no_successor = true;
135  result.dead_state = current.s;
136  result.impossible_successors = next_symbols;
137  return result;
138  }
139  }
140  }
141  }
142 
143 #ifdef DEBUG_PRINTING
144  if (DEBUG_PRINTING >= 2) {
145  std::cout << "States discovered: " << reachable.size() << std::endl;
146  std::cout << "Marking predecessors" << std::endl;
147  }
148 #endif
149 
150  // marking the predecessors
151  std::deque<State> pfrontier(final_states.begin(), final_states.end());
152  std::unordered_set<State> good_states; // states that terminate
153 
154  while (!pfrontier.empty()) {
155  const State current = pfrontier.front();
156  pfrontier.pop_front();
157  if (good_states.find(current)==good_states.end()) {
158  good_states.insert(current);
159  for (State pred : predecessor_map[current]) {
160  pfrontier.push_back(pred);
161  }
162  }
163  }
164 
165 #ifdef DEBUG_PRINTING
166  if (DEBUG_PRINTING >= 2) {
167  std::cout << "Finding non-terminating states" << std::endl;
168  }
169 #endif
170 
171  // now go through the state space once again to find a state that is not marked
172  std::deque<frontier_item> frontier2 = initial_states_conv(initial_states);
173  std::unordered_set<State> seen;
174  while (!frontier2.empty()) {
175  const frontier_item current = frontier2.front();
176  frontier2.pop_front();
177  if (seen.insert(current.s).second) {
178  Symbol_set next_symbols = automaton.next_symbols(current.s);
179 
180  for (Symbol sigma : next_symbols) {
181  State_set states = automaton.successors(current.s, sigma);
182  for (State state : states) {
183  frontier_item next(state, current.cex_chain, sigma);
184  frontier2.push_front(next);
185  }
186  }
187 
188 
189  if (good_states.find(current.s) == good_states.end()) {
190 
191  // find the loop from this state
192  std::deque<frontier_item> frontier3;
193  std::unordered_set<State> seen3;
194  frontier3.push_back(frontier_item(current.s));
195  while (!frontier3.empty()) {
196  const frontier_item current3 = frontier3.front();
197  frontier3.pop_front();
198  if (seen3.insert(current3.s).second) {
199  Symbol_set next_symbols = automaton.next_symbols(current3.s);
200 
201  for (Symbol sigma : next_symbols) {
202  State_set states = automaton.successors(current3.s, sigma);
203  for (State state : states) {
204  frontier_item next(state, current3.cex_chain, sigma);
205  frontier3.push_back(next);
206  }
207  }
208  } else {
209  // we found the loop
210  result.loop = current3.cex_chain->to_list();
211  }
212  }
213 
214  result.counter_example = current.cex_chain->to_list();
215  result.deadlock_found = true;
216  result.no_successor = false;
217  result.dead_state = current.s;
218  result.impossible_successors = next_symbols;
219  return result;
220  }
221  }
222  }
223 
224  return result;
225  }
