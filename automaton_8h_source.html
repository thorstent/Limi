336  void filter_epsilon(Symbol_set& symbols) const {
337  if (collapse_epsilon) {
338  for (auto it = symbols.begin(); it!=symbols.end(); ) {

347  void explore_epsilon(State_set& states) const {
348  if (!collapse_epsilon) return;
349  State_set seen;
350  std::deque<State> frontier;
351  frontier.insert(frontier.begin(), states.begin(), states.end());
352  states.clear();
353  while(!frontier.empty()) {
354  const State& s = frontier.front();
355  //state_printer()(s, std::cout); std::cout << std::endl;
356 
357  Symbol_set next_symbols;
358  impl().int_next_symbols(s, next_symbols);
359  State_set succs;
360  bool first = true;
361  if (impl().int_is_final_state(s)) {
362  first = false;
363  states.insert(s);
364  }
365  for (const Symbol& sy : next_symbols) {
366  if (impl().int_is_epsilon(sy)) {
367  impl().int_successors(s, sy, succs);
368  } else if (first) { // if there is at least one non-epsilon transition we add this state
369  first = false;
370  states.insert(s);
371  }
372  }
373  for (const State& st : succs) {
374  if (seen.insert(st).second) {
375  frontier.push_back(st);
376  }
377  }
378  frontier.pop_front();
379  }
380  //std::cout << std::endl;
381  }
