101  void prune(std::shared_ptr<StateB_set>& b, StateBI_set& un_pruned, unsigned k, bool dirty) {
102  for(auto it = b->begin(); it!=b->end(); ) {
103  if ((**it).size() > k) {
104  if (!un_pruned && !dirty) {
105  un_pruned = std::make_shared<StateB_set>(*b);
106  }
107  it = b->erase(it);

109  ++it;

197  inclusion_result<Symbol> run()
198  {
199  inclusion_result<Symbol> result;
200  result.included = true;
201  result.bound_hit = false;
202  result.max_bound = bound;
203 
204 #ifdef DEBUG_PRINTING
205  unsigned loop_counter = 0;
206  unsigned transitions = 0;
207 #endif
208  while (frontier.size() > 0) {
209 #ifdef DEBUG_PRINTING
210  if (DEBUG_PRINTING>=2 && loop_counter % 1000 == 0) std::cout << loop_counter << " rounds; A states: " << antichain.size() << std::endl;
211 #endif
212  pair current = frontier.top();
213  frontier.pop();
214 
215  Symbol_set next_symbols;
216  a.next_symbols(current.a, next_symbols);
217 
218 #ifdef DEBUG_PRINTING
219  ++ loop_counter;
220 #endif
221  if ((a.is_final_state(current.a) && !b.is_final_state(*current.b))) {
222  result.counter_example = current.cex_chain->to_list();
223  result.included = false;
224  if (current.dirty)
225  result.bound_hit = true;
226  break;
227  }
228 
229 #ifdef DEBUG_PRINTING
230  if (DEBUG_PRINTING>=3) {
231  debug << "Next pair: ";
232  debug << a.state_printer()(current.a) << " - ";
233  if (DEBUG_PRINTING>=5) {
234  internal::print_set(*current.b, debug, b.state_printer());
235  }
236  debug << std::endl;
237  }
238 #endif
239 
240  for (Symbol sigma : next_symbols) {
241 #ifdef DEBUG_PRINTING
242  ++transitions;
243  if (DEBUG_PRINTING>=4) {
244  std::cout << "Symbol: ";
245  std::cout << a.symbol_printer()(sigma);
246  std::cout << std::endl;
247  }
248 #endif
249  StateA_set states_a = a.successors(current.a, sigma);
250  StateBI_set unpruned;
251  StateBI_set states_b;
252  if (b.is_epsilon(sigma)) states_b=current.b; else {
253  auto states_b1 = std::make_shared<StateB_set>();
254  b.successors(*current.b, sigma, *states_b1);
255  prune(states_b1, unpruned, bound, current.dirty);
256  states_b = states_b1;
257  }
258 
259  for (StateA state_a : states_a) {
260  antichain_algo_ind::pair next(state_a, states_b, current.cex_chain, sigma);
261  next.dirty = current.dirty || unpruned;
262  if (unpruned) before_dirty.push_back(antichain_algo_ind::pair(state_a, unpruned, current.cex_chain, sigma));
263 
264  if (!antichain.contains(next.a, next.b)) {
265  antichain.add(next.a, next.b, next.dirty);
266  frontier.push(std::move(next));
267  }
268  }
269  }
270 
271  }
272 
273 #ifdef DEBUG_PRINTING
274  if (DEBUG_PRINTING>=1) debug << loop_counter << " rounds; seen states: " << antichain.size() << "; transitions: " << transitions << std::endl;
275 
276  if (DEBUG_PRINTING >= 4) {
277  antichain.print(debug, a.state_printer(), b.state_printer());
278  }
279 #endif
280 
281  return result;
282  }
