193  if (!deadlock_found)
194  stream << "No deadlock found" << std::endl;
195  else {
196  stream << "Deadlock";
197  stream << std::endl;
198  if (no_successor) stream << "Dead state (no successor)"; else stream << "No final state is reachable from";
199  stream << ": " << state_printer(dead_state) << std::endl;
200  if (no_successor) {
201  if (!impossible_successors.empty()) {
202  stream << "Impossible successors:" << std::endl;
203  for (const auto& sy : impossible_successors) {
204  stream << symbol_printer(sy);
205  stream << std::endl;
206  }
207  } else {
208  stream << "No successors at all." << std::endl;
209  }
210  } else {
211  if (!loop.empty()) {
212  stream << "Loop:" << std::endl;
213  for (const auto& sy : loop) {
214  stream << symbol_printer(sy);
215  stream << std::endl;
216  }
217  }
218  }
219  stream << "Trace:" << std::endl;
220  for (const auto& s : counter_example) {
221  stream << symbol_printer(s);
222  stream << std::endl;
223  }
224  }
225  }
